4//! # Geographic Location Detection
5//!
6//! Automatically detects user's geographic location for optimal CDN selection.
7//! Uses multiple detection methods with fallback strategies.
8
9use serde::{Deserialize, Serialize};
10use std::time::Duration;
11use tokio::time::timeout;
12use tracing::{debug, info, warn};
13
14use crate::config::Region;
15use crate::error::{Result, TurboCdnError};
16
17/// Geographic location detector
18#[derive(Debug)]
19pub struct GeoDetector {
20    client: reqwest::Client,
21    cache: Option<DetectionResult>,
22    cache_ttl: Duration,
23}
24
25/// Detection result with confidence score
26#[derive(Debug, Clone, Serialize, Deserialize)]
27pub struct DetectionResult {
28    pub region: Region,
29    pub country_code: String,
30    pub confidence: f64,
31    pub detected_at: chrono::DateTime<chrono::Utc>,
32    pub method: DetectionMethod,
33}
34
35/// Detection method used
36#[derive(Debug, Clone, Serialize, Deserialize)]
37pub enum DetectionMethod {
38    IpApi,
39    NetworkTest,
40    Fallback,
41    Manual,
42}
43
44/// IP geolocation response from ip-api.com
45#[derive(Debug, Deserialize)]
46struct IpApiResponse {
47    status: String,
48    country: String,
49    #[serde(rename = "countryCode")]
50    country_code: String,
51    #[allow(dead_code)]
52    region: String,
53    #[serde(rename = "regionName")]
54    #[allow(dead_code)]
55    region_name: String,
56    #[allow(dead_code)]
57    city: String,
58    timezone: String,
59}
60
61impl GeoDetector {
62    /// Create a new geo detector
63    pub fn new() -> Self {
64        let client = reqwest::Client::builder()
65            .timeout(Duration::from_secs(10))
66            .user_agent("turbo-cdn/1.0.0")
67            .build()
68            .unwrap_or_else(|_| reqwest::Client::new());
69
70        Self {
71            client,
72            cache: None,
73            cache_ttl: Duration::from_secs(24 * 60 * 60), // 24 hours
74        }
75    }
76
77    /// Detect user's geographic region
78    pub async fn detect_region(&mut self) -> Result<Region> {
79        // Check cache first
80        if let Some(cached) = &self.cache {
81            let age = chrono::Utc::now().signed_duration_since(cached.detected_at);
82            if age.to_std().unwrap_or(Duration::MAX) < self.cache_ttl {
83                debug!("Using cached geo detection result: {:?}", cached.region);
84                return Ok(cached.region.clone());
85            }
86        }
87
88        info!("Detecting geographic location...");
89
90        // Try multiple detection methods
91        let result = self.detect_with_fallback().await?;
92
93        // Cache the result
94        self.cache = Some(result.clone());
95
96        info!(
97            "Detected region: {:?} (confidence: {:.1}%, method: {:?})",
98            result.region,
99            result.confidence * 100.0,
100            result.method
101        );
102
103        Ok(result.region)
104    }
105
106    /// Detect with multiple fallback methods
107    async fn detect_with_fallback(&self) -> Result<DetectionResult> {
108        // Method 1: IP geolocation API
109        if let Ok(result) = self.detect_via_ip_api().await {
110            return Ok(result);
111        }
112
113        warn!("IP geolocation failed, trying network tests...");
114
115        // Method 2: Network performance tests
116        if let Ok(result) = self.detect_via_network_test().await {
117            return Ok(result);
118        }
119
120        warn!("Network tests failed, using fallback...");
121
122        // Method 3: Fallback to Global
123        Ok(DetectionResult {
124            region: Region::Global,
125            country_code: "UNKNOWN".to_string(),
126            confidence: 0.1,
127            detected_at: chrono::Utc::now(),
128            method: DetectionMethod::Fallback,
129        })
130    }
131
132    /// Detect region using IP geolocation API
133    async fn detect_via_ip_api(&self) -> Result<DetectionResult> {
134        let url = "http://ip-api.com/json/?fields=status,country,countryCode,region,regionName,city,timezone";
135
136        let response = timeout(Duration::from_secs(5), self.client.get(url).send())
137            .await
138            .map_err(|_| TurboCdnError::network("IP geolocation request timeout"))?
139            .map_err(TurboCdnError::Network)?;
140
141        let ip_info: IpApiResponse = response.json().await.map_err(TurboCdnError::Network)?;
142
143        if ip_info.status != "success" {
144            return Err(TurboCdnError::network("IP geolocation API failed"));
145        }
146
147        let region = self.country_to_region(&ip_info.country_code);
148        let confidence = self.calculate_confidence(&ip_info);
149
150        debug!(
151            "IP geolocation: {} ({}) -> {:?}",
152            ip_info.country, ip_info.country_code, region
153        );
154
155        Ok(DetectionResult {
156            region,
157            country_code: ip_info.country_code,
158            confidence,
159            detected_at: chrono::Utc::now(),
160            method: DetectionMethod::IpApi,
161        })
162    }
163
164    /// Detect region using network performance tests
165    async fn detect_via_network_test(&self) -> Result<DetectionResult> {
166        let test_urls = vec![
167            ("github", "https://api.github.com"),
168            ("jsdelivr", "https://cdn.jsdelivr.net"),
169            ("fastly", "https://fastly.jsdelivr.net"),
170            ("cloudflare", "https://cdnjs.cloudflare.com"),
171        ];
172
173        let mut results = Vec::new();
174
175        for (name, url) in test_urls {
176            if let Ok(latency) = self.test_latency(url).await {
177                results.push((name, latency));
178                debug!("Network test {}: {:.0}ms", name, latency);
179            }
180        }
181
182        if results.is_empty() {
183            return Err(TurboCdnError::network("All network tests failed"));
184        }
185
186        // Sort by latency
187        results.sort_by(|a, b| a.1.partial_cmp(&b.1).unwrap_or(std::cmp::Ordering::Equal));
188
189        let best_cdn = results[0].0;
190        let region = self.cdn_to_region(best_cdn);
191
192        Ok(DetectionResult {
193            region,
194            country_code: "DETECTED".to_string(),
195            confidence: 0.7,
196            detected_at: chrono::Utc::now(),
197            method: DetectionMethod::NetworkTest,
198        })
199    }
200
201    /// Test latency to a URL
202    async fn test_latency(&self, url: &str) -> Result<f64> {
203        let start = std::time::Instant::now();
204
205        let _response = timeout(Duration::from_secs(3), self.client.head(url).send())
206            .await
207            .map_err(|_| TurboCdnError::network("Latency test timeout"))?
208            .map_err(TurboCdnError::Network)?;
209
210        let latency = start.elapsed().as_millis() as f64;
211        Ok(latency)
212    }

214    /// Map country code to region
215    fn country_to_region(&self, country_code: &str) -> Region {
216        match country_code {
217            // China
218            "CN" => Region::China,
219
220            // North America
221            "US" | "CA" | "MX" => Region::NorthAmerica,
222
223            // Europe
224            "GB" | "DE" | "FR" | "IT" | "ES" | "NL" | "SE" | "NO" | "DK" | "FI" | "PL" | "CZ"
225            | "AT" | "CH" | "BE" | "IE" | "PT" | "GR" | "HU" | "RO" | "BG" | "HR" | "SI" | "SK"
226            | "LT" | "LV" | "EE" | "LU" | "MT" | "CY" => Region::Europe,
227
228            // Asia Pacific
229            "JP" | "KR" | "SG" | "HK" | "TW" | "AU" | "NZ" | "IN" | "TH" | "MY" | "ID" | "PH"
230            | "VN" | "BD" | "PK" | "LK" | "MM" | "KH" | "LA" | "BN" => Region::AsiaPacific,
231
232            // Default to Global for other countries
233            _ => Region::Global,
234        }
235    }
236
237    /// Map best performing CDN to likely region
238    fn cdn_to_region(&self, cdn: &str) -> Region {
239        match cdn {
240            "github" => Region::NorthAmerica, // GitHub is US-based
241            "jsdelivr" => Region::Europe,     // jsDelivr has good EU presence
242            "fastly" => Region::China,        // Fastly works well in China
243            "cloudflare" => Region::Global,   // Cloudflare is global
244            _ => Region::Global,
245        }
246    }
247
248    /// Calculate confidence based on IP info quality
249    fn calculate_confidence(&self, ip_info: &IpApiResponse) -> f64 {
250        let mut confidence: f64 = 0.8; // Base confidence for IP geolocation
251
252        // Higher confidence for specific countries
253        if matches!(
254            ip_info.country_code.as_str(),
255            "CN" | "US" | "GB" | "DE" | "JP"
256        ) {
257            confidence += 0.1;
258        }
259
260        // Lower confidence if timezone doesn't match expected region
261        if ip_info.timezone.is_empty() {
262            confidence -= 0.1;
263        }
264
265        confidence.clamp(0.1, 0.95)
266    }
267
268    /// Get cached detection result
269    pub fn get_cached_result(&self) -> Option<&DetectionResult> {
270        self.cache.as_ref()
271    }
272
273    /// Clear cache to force re-detection
274    pub fn clear_cache(&mut self) {
275        self.cache = None;
276    }
277}
278
279impl Default for GeoDetector {
280    fn default() -> Self {
281        Self::new()
282    }
