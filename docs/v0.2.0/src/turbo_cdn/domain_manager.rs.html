4//! # Domain Manager
5//!
6//! Extensible domain management system for turbo-cdn.
7//! Provides flexible domain validation with support for:
8//! - Built-in trusted domains
9//! - Custom domain lists
10//! - Domain categories and tags
11//! - Runtime domain management
12//! - Validation bypass options
13
14use crate::error::{Result, TurboCdnError};
15use serde::{Deserialize, Serialize};
16use std::collections::HashMap;
17
18/// Domain category for organizing trusted domains
19#[derive(Debug, Clone, PartialEq, Eq, Hash, Serialize, Deserialize)]
20pub enum DomainCategory {
21    /// Official package repositories
22    PackageRegistry,
23    /// Content delivery networks
24    CDN,
25    /// Version control systems
26    VersionControl,
27    /// Official language/framework sites
28    Official,
29    /// Archive and mirror sites
30    Archive,
31    /// Custom user-defined category
32    Custom(String),
33}
34
35/// Domain entry with metadata
36#[derive(Debug, Clone, Serialize, Deserialize)]
37pub struct DomainEntry {
38    /// Domain name (e.g., "github.com")
39    pub domain: String,
40    /// Category classification
41    pub category: DomainCategory,
42    /// Optional description
43    pub description: Option<String>,
44    /// Tags for flexible filtering
45    pub tags: Vec<String>,
46    /// Whether this domain is enabled
47    pub enabled: bool,
48    /// Trust level (0-100, higher = more trusted)
49    pub trust_level: u8,
50}
51
52/// Domain validation configuration
53#[derive(Debug, Clone, Serialize, Deserialize)]
54pub struct DomainValidationConfig {
55    /// Enable domain validation
56    pub enabled: bool,
57    /// Strict mode - only allow explicitly listed domains
58    pub strict_mode: bool,
59    /// Allow subdomains of trusted domains
60    pub allow_subdomains: bool,
61    /// Minimum trust level required (0-100)
62    pub min_trust_level: u8,
63    /// Categories to allow (empty = allow all)
64    pub allowed_categories: Vec<DomainCategory>,
65    /// Tags to allow (empty = allow all)
66    pub allowed_tags: Vec<String>,
67}
68
69/// Domain manager for extensible domain validation
70#[derive(Debug, Clone)]
71pub struct DomainManager {
72    /// Domain entries by domain name
73    domains: HashMap<String, DomainEntry>,
74    /// Validation configuration
75    config: DomainValidationConfig,
76}

99        self.tags = tags;

101    }
102

105        self.trust_level = trust_level.min(100);
106        self
107    }
108
109    /// Enable or disable domain
110    pub fn with_enabled(mut self, enabled: bool) -> Self {
111        self.enabled = enabled;
112        self
113    }

315    /// Add a custom domain
316    pub fn add_domain(&mut self, domain: DomainEntry) -> Result<()> {
317        if domain.domain.is_empty() {
318            return Err(TurboCdnError::config(
319                "Domain name cannot be empty".to_string(),
320            ));
321        }
322
323        self.domains.insert(domain.domain.clone(), domain);
324        Ok(())
325    }

389    /// Validate a URL against domain rules
390    pub fn validate_url(&self, url: &str) -> Result<()> {
391        // If validation is disabled, allow everything
392        if !self.config.enabled {
393            return Ok(());
394        }
395
396        let parsed_url = url::Url::parse(url)
397            .map_err(|e| TurboCdnError::compliance(format!("Invalid URL: {}", e)))?;
398
399        let domain = parsed_url.domain().ok_or_else(|| {
400            TurboCdnError::compliance("URL does not have a valid domain".to_string())
401        })?;
402
403        self.validate_domain(domain)
404    }

406    /// Validate a domain against rules
407    pub fn validate_domain(&self, domain: &str) -> Result<()> {
408        // If validation is disabled, allow everything
409        if !self.config.enabled {
410            return Ok(());
411        }
412
413        // Try exact match first
414        if let Some(entry) = self.domains.get(domain) {
415            return self.validate_domain_entry(entry);
416        }
417
418        // If subdomain checking is enabled, check parent domains
419        if self.config.allow_subdomains {
420            for (trusted_domain, entry) in &self.domains {
421                if domain.ends_with(&format!(".{}", trusted_domain)) {
422                    return self.validate_domain_entry(entry);
423                }
424            }
425        }
426
427        // In strict mode, reject unknown domains
428        if self.config.strict_mode {
429            return Err(TurboCdnError::compliance(format!(
430                "Domain '{}' is not in the allowed list",
431                domain
432            )));
433        }
434
435        // In non-strict mode, allow unknown domains
436        Ok(())
437    }

439    /// Validate a domain entry against configuration rules
440    fn validate_domain_entry(&self, entry: &DomainEntry) -> Result<()> {
441        // Check if domain is enabled
442        if !entry.enabled {
443            return Err(TurboCdnError::compliance(format!(
444                "Domain '{}' is disabled",
445                entry.domain
446            )));
447        }
448
449        // Check trust level
450        if entry.trust_level < self.config.min_trust_level {
451            return Err(TurboCdnError::compliance(format!(
452                "Domain '{}' trust level ({}) below minimum required ({})",
453                entry.domain, entry.trust_level, self.config.min_trust_level
454            )));
455        }
456
457        // Check allowed categories (if specified)
458        if !self.config.allowed_categories.is_empty()
459            && !self.config.allowed_categories.contains(&entry.category)
460        {
461            return Err(TurboCdnError::compliance(format!(
462                "Domain '{}' category '{:?}' not in allowed categories",
463                entry.domain, entry.category
464            )));
465        }
466
467        // Check allowed tags (if specified)
468        if !self.config.allowed_tags.is_empty() {
469            let has_allowed_tag = entry
470                .tags
471                .iter()
472                .any(|tag| self.config.allowed_tags.contains(tag));
473
474            if !has_allowed_tag {
475                return Err(TurboCdnError::compliance(format!(
476                    "Domain '{}' does not have any allowed tags",
477                    entry.domain
478                )));
479            }
480        }
481
482        Ok(())
483    }

485    /// Get statistics about managed domains
486    pub fn get_stats(&self) -> DomainStats {
487        let total = self.domains.len();
488        let enabled = self.domains.values().filter(|d| d.enabled).count();
489        let disabled = total - enabled;
490
491        let mut by_category = HashMap::new();
492        let mut by_trust_level = HashMap::new();
493
494        for entry in self.domains.values() {
495            *by_category.entry(entry.category.clone()).or_insert(0) += 1;
496
497            let trust_range = match entry.trust_level {
498                90..=100 => "High (90-100)".to_string(),
499                70..=89 => "Medium (70-89)".to_string(),
500                50..=69 => "Low (50-69)".to_string(),
501                _ => "Very Low (0-49)".to_string(),
502            };
503            *by_trust_level.entry(trust_range).or_insert(0) += 1;
504        }
505
506        DomainStats {
507            total,
508            enabled,
509            disabled,
510            by_category,
511            by_trust_level,
512        }
513    }
