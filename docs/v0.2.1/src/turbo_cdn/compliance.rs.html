159    /// Check if a download request is compliant
160    pub async fn check_compliance(&self, request: &DownloadRequest) -> Result<ComplianceResult> {
161        let mut result = ComplianceResult {
162            request_id: request.id,
163            approved: true,
164            reasons: Vec::new(),
165            license_info: None,
166            risk_level: RiskLevel::Low,
167            timestamp: chrono::Utc::now(),
168        };
169
170        // Check user consent if required
171        if self.config.data_protection.user_consent_required && !request.user_consent {
172            result.approved = false;
173            result
174                .reasons
175                .push("User consent required but not provided".to_string());
176            result.risk_level = RiskLevel::Critical;
177        }
178
179        // Validate source domain using domain manager
180        if self.config.validate_source {
181            if let Err(e) = self.domain_manager.validate_url(&request.url) {
182                result.approved = false;
183                result
184                    .reasons
185                    .push(format!("Domain validation failed: {}", e));
186                result.risk_level = std::cmp::max(result.risk_level, RiskLevel::High);
187            }
188        }
189
190        // Check for blocked content patterns
191        if let Err(e) = self
192            .content_validator
193            .check_blocked_patterns(&request.url, &request.file_name)
194        {
195            result.approved = false;
196            result
197                .reasons
198                .push(format!("Content validation failed: {}", e));
199            result.risk_level = RiskLevel::Critical;
200        }
201
202        // Verify open source license if repository is provided
203        if self.config.verify_open_source {
204            if let Some(repo) = &request.repository {
205                match self.license_validator.validate_repository(repo).await {
206                    Ok(license_info) => {
207                        if !license_info.is_open_source {
208                            result.approved = false;
209                            result.reasons.push(
210                                "Repository does not have an open source license".to_string(),
211                            );
212                            result.risk_level = RiskLevel::Critical;
213                        }
214                        result.license_info = Some(license_info);
215                    }
216                    Err(e) => {
217                        warn!("Failed to validate license for repository {}: {}", repo, e);
218                        if self.config.strict_mode {
219                            result.approved = false;
220                            result
221                                .reasons
222                                .push(format!("License validation failed: {}", e));
223                            result.risk_level = std::cmp::max(result.risk_level, RiskLevel::Medium);
224                        }
225                    }
226                }
227            }
228        }
229
230        // Log the compliance check
231        if self.config.audit_logging {
232            let audit_entry = AuditLogEntry {
233                id: Uuid::new_v4(),
234                timestamp: chrono::Utc::now(),
235                event_type: AuditEventType::ComplianceCheck,
236                request: request.clone(),
237                result: result.clone(),
238                additional_data: serde_json::Value::Null,
239            };
240
241            if let Err(e) = self.audit_logger.log_entry(&audit_entry).await {
242                warn!("Failed to log audit entry: {}", e);
243            }
244        }
245
246        if result.approved {
247            info!("Compliance check passed for request {}", request.id);
248        } else {
249            warn!(
250                "Compliance check failed for request {}: {:?}",
251                request.id, result.reasons
252            );
253        }
254
255        Ok(result)
256    }

258    /// Log a download event
259    pub async fn log_download_event(
260        &self,
261        event_type: AuditEventType,
262        request: &DownloadRequest,
263        additional_data: Option<serde_json::Value>,
264    ) -> Result<()> {
265        if !self.config.audit_logging {
266            return Ok(());
267        }
268
269        let result = ComplianceResult {
270            request_id: request.id,
271            approved: true,
272            reasons: Vec::new(),
273            license_info: None,
274            risk_level: RiskLevel::Low,
275            timestamp: chrono::Utc::now(),
276        };
277
278        let audit_entry = AuditLogEntry {
279            id: Uuid::new_v4(),
280            timestamp: chrono::Utc::now(),
281            event_type,
282            request: request.clone(),
283            result,
284            additional_data: additional_data.unwrap_or(serde_json::Value::Null),
285        };
286
287        self.audit_logger.log_entry(&audit_entry).await
288    }

291impl AuditLogger {
292    fn new(log_path_str: Option<&str>, enabled: bool) -> Result<Self> {
293        let path_manager = PathManager::default();
294
295        // Get audit log path - use provided path or default to data directory
296        let log_path = if let Some(path_str) = log_path_str {
297            // Expand the provided path (handles ~ and environment variables)
298            path_manager.expand_path(path_str)?
299        } else {
300            // Use default audit log path in data directory
301            path_manager.data_file("audit.log")?
302        };
303
304        if enabled {
305            // Ensure the parent directory exists using PathManager
306            if let Some(parent) = log_path.parent() {
307                // Use sync version since this is called from sync context
308                std::fs::create_dir_all(parent).map_err(|e| {
309                    TurboCdnError::compliance(format!(
310                        "Failed to create audit log directory: {}",
311                        e
312                    ))
313                })?;
314            }
315        }
316
317        Ok(Self { log_path, enabled })
318    }
319
320    async fn log_entry(&self, entry: &AuditLogEntry) -> Result<()> {
321        if !self.enabled {
322            return Ok(());
323        }
324
325        let json_entry = serde_json::to_string(entry).map_err(|e| {
326            TurboCdnError::compliance(format!("Failed to serialize audit entry: {}", e))
327        })?;
328
329        let log_line = format!("{}\n", json_entry);
330
331        tokio::fs::OpenOptions::new()
332            .create(true)
333            .append(true)
334            .open(&self.log_path)
335            .await
336            .map_err(|e| {
337                TurboCdnError::compliance(format!("Failed to open audit log file: {}", e))
338            })?
339            .write_all(log_line.as_bytes())
340            .await
341            .map_err(|e| {
342                TurboCdnError::compliance(format!("Failed to write audit log entry: {}", e))
343            })?;
344
345        debug!("Logged audit entry: {}", entry.id);
346        Ok(())
347    }
348}

350impl LicenseValidator {
351    fn new() -> Self {
352        let mut known_open_source_licenses = HashSet::new();
353
354        // Add common open source licenses
355        let licenses = vec![
356            "MIT",
357            "Apache-2.0",
358            "GPL-3.0",
359            "GPL-2.0",
360            "BSD-3-Clause",
361            "BSD-2-Clause",
362            "ISC",
363            "MPL-2.0",
364            "LGPL-3.0",
365            "LGPL-2.1",
366            "AGPL-3.0",
367            "Unlicense",
368            "CC0-1.0",
369            "Zlib",
370            "BSL-1.0",
371            "WTFPL",
372            "0BSD",
373        ];
374
375        for license in licenses {
376            known_open_source_licenses.insert(license.to_string());
377        }
378
379        Self {
380            known_open_source_licenses,
381        }
382    }
383
384    async fn validate_repository(&self, repository: &str) -> Result<LicenseInfo> {
385        // For GitHub repositories, we can check the license via API
386        if repository.contains("github.com") {
387            return self.validate_github_repository(repository).await;
388        }
389
390        // For other repositories, we'll do a basic check
391        self.validate_generic_repository(repository).await
392    }
393
394    async fn validate_github_repository(&self, repository: &str) -> Result<LicenseInfo> {
395        // Extract owner/repo from URL
396        let repo_path = repository
397            .trim_start_matches("https://github.com/")
398            .trim_start_matches("http://github.com/")
399            .trim_end_matches(".git")
400            .trim_end_matches('/');
401
402        let api_url = format!("https://api.github.com/repos/{}", repo_path);
403
404        let client = reqwest::Client::new();
405        let response = client
406            .get(&api_url)
407            .header("User-Agent", "turbo-cdn")
408            .send()
409            .await
410            .map_err(|e| {
411                TurboCdnError::compliance(format!("Failed to fetch repository info: {}", e))
412            })?;
413
414        if !response.status().is_success() {
415            return Err(TurboCdnError::compliance(format!(
416                "GitHub API returned status: {}",
417                response.status()
418            )));
419        }
420
421        let repo_info: serde_json::Value = response.json().await.map_err(|e| {
422            TurboCdnError::compliance(format!("Failed to parse GitHub API response: {}", e))
423        })?;
424
425        let license_info = repo_info.get("license");
426
427        if let Some(license) = license_info {
428            if license.is_null() {
429                return Ok(LicenseInfo {
430                    license_type: "Unknown".to_string(),
431                    is_open_source: false,
432                    allows_redistribution: false,
433                    requires_attribution: false,
434                    source_url: Some(repository.to_string()),
435                });
436            }
437
438            let license_key = license
439                .get("key")
440                .and_then(|k| k.as_str())
441                .unwrap_or("unknown");
442
443            let license_name = license
444                .get("name")
445                .and_then(|n| n.as_str())
446                .unwrap_or("Unknown");
447
448            let is_open_source = self.known_open_source_licenses.contains(license_key)
449                || self.known_open_source_licenses.contains(license_name);
450
451            Ok(LicenseInfo {
452                license_type: license_name.to_string(),
453                is_open_source,
454                allows_redistribution: is_open_source,
455                requires_attribution: is_open_source
456                    && !matches!(license_key, "unlicense" | "cc0-1.0"),
457                source_url: Some(repository.to_string()),
458            })
459        } else {
460            Ok(LicenseInfo {
461                license_type: "Unknown".to_string(),
462                is_open_source: false,
463                allows_redistribution: false,
464                requires_attribution: false,
465                source_url: Some(repository.to_string()),
466            })
467        }
468    }
469
470    async fn validate_generic_repository(&self, repository: &str) -> Result<LicenseInfo> {
471        // For non-GitHub repositories, we'll be more conservative
472        // and assume they're not open source unless we can verify otherwise
473        Ok(LicenseInfo {
474            license_type: "Unknown".to_string(),
475            is_open_source: false,
476            allows_redistribution: false,
477            requires_attribution: false,
478            source_url: Some(repository.to_string()),
479        })
480    }
481}

483impl ContentValidator {
484    fn new() -> Result<Self> {
485        let mut blocked_patterns = Vec::new();
486
487        // Add patterns for potentially problematic content
488        let patterns = vec![
489            r"(?i)crack",
490            r"(?i)keygen",
491            r"(?i)serial",
492            r"(?i)patch",
493            r"(?i)pirate",
494            r"(?i)warez",
495            r"(?i)nulled",
496            r"(?i)torrent",
497        ];
498
499        for pattern in patterns {
500            let regex = regex::Regex::new(pattern)
501                .map_err(|e| TurboCdnError::compliance(format!("Invalid regex pattern: {}", e)))?;
502            blocked_patterns.push(regex);
503        }
504
505        let mut allowed_domains = HashSet::new();
506        let domains = vec![
507            "github.com",
508            "githubusercontent.com",
509            "cdn.jsdelivr.net",
510            "fastly.jsdelivr.net",
511            "cdnjs.cloudflare.com",
512            "unpkg.com",
513            "registry.npmjs.org",
514            "pypi.org",
515            "crates.io",
516            "maven.org",
517            "repo1.maven.org",
518            "central.sonatype.com",
519        ];
520
521        for domain in domains {
522            allowed_domains.insert(domain.to_string());
523        }
524
525        Ok(Self {
526            blocked_patterns,
527            allowed_domains,
528        })
529    }
530
531    #[allow(dead_code)]
532    fn validate_source(&self, url: &str) -> Result<()> {
533        let parsed_url = url::Url::parse(url)
534            .map_err(|e| TurboCdnError::compliance(format!("Invalid URL: {}", e)))?;
535
536        if let Some(domain) = parsed_url.domain() {
537            if !self
538                .allowed_domains
539                .iter()
540                .any(|allowed| domain.ends_with(allowed))
541            {
542                return Err(TurboCdnError::compliance(format!(
543                    "Domain '{}' is not in the allowed list",
544                    domain
545                )));
546            }
547        } else {
548            return Err(TurboCdnError::compliance(
549                "URL does not have a valid domain".to_string(),
550            ));
551        }
552
553        Ok(())
554    }
555
556    fn check_blocked_patterns(&self, url: &str, file_name: &str) -> Result<()> {
557        let combined_text = format!("{} {}", url, file_name);
558
559        for pattern in &self.blocked_patterns {
560            if pattern.is_match(&combined_text) {
561                return Err(TurboCdnError::compliance(format!(
562                    "Content matches blocked pattern: {}",
563                    pattern.as_str()
564                )));
565            }
566        }
567
568        Ok(())
569    }

572impl std::cmp::PartialOrd for RiskLevel {
573    fn partial_cmp(&self, other: &Self) -> Option<std::cmp::Ordering> {
574        Some(self.cmp(other))
575    }
576}

578impl std::cmp::Ord for RiskLevel {
579    fn cmp(&self, other: &Self) -> std::cmp::Ordering {
580        let self_value = match self {
581            RiskLevel::Low => 0,
582            RiskLevel::Medium => 1,
583            RiskLevel::High => 2,
584            RiskLevel::Critical => 3,
585        };
586
587        let other_value = match other {
588            RiskLevel::Low => 0,
589            RiskLevel::Medium => 1,
590            RiskLevel::High => 2,
591            RiskLevel::Critical => 3,
592        };
593
594        self_value.cmp(&other_value)
595    }
