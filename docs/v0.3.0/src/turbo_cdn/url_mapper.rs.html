4//! URL mapping system for CDN optimization
5//!
6//! This module provides automatic URL mapping from original sources to optimized CDN URLs
7//! based on regex patterns and geographic location.
8

20/// URL mapping rule configuration
21#[derive(Debug, Clone)]
22pub struct UrlMappingRule {
23    /// Regex pattern to match URLs
24    pub pattern: Regex,
25    /// Replacement URL templates (in priority order)
26    pub replacements: Vec<String>,
27    /// Applicable regions for this rule
28    pub regions: Vec<Region>,
29    /// Priority (lower = higher priority)
30    pub priority: u32,
31    /// Whether this rule is enabled
32    pub enabled: bool,
33}
34
35/// Cache entry for URL mappings
36#[derive(Debug, Clone)]
37struct CacheEntry {
38    /// Mapped URLs
39    urls: Vec<String>,
40    /// Timestamp when cached
41    cached_at: Instant,
42    /// TTL for this entry
43    ttl: Duration,
44}
45
46impl CacheEntry {
47    fn new(urls: Vec<String>, ttl: Duration) -> Self {
48        Self {
49            urls,
50            cached_at: Instant::now(),
51            ttl,
52        }
53    }
54
55    fn is_expired(&self) -> bool {
56        self.cached_at.elapsed() > self.ttl
57    }
58}
59
60/// High-performance concurrent cache
61type CacheMap = DashMap<String, CacheEntry>;
62
63/// URL mapper for CDN optimization with performance-based selection
64#[derive(Debug)]
65pub struct UrlMapper {
66    rules: Vec<UrlMappingRule>,
67    current_region: Region,
68    cache: CacheMap,
69    cache_enabled: bool,
70    cache_ttl: Duration,
71    max_cache_entries: usize,
72    server_tracker: Arc<Mutex<ServerTracker>>,
73    #[allow(dead_code)]
74    quality_assessor: Option<Arc<CdnQualityAssessor>>,
75}
76
77impl UrlMapper {
78    /// Create high-performance cache
79    fn create_cache() -> CacheMap {
80        DashMap::new()
81    }
82
83    /// Create a new URL mapper with configuration
84    pub fn new(config: &TurboCdnConfig, region: Region) -> Result<Self> {
85        let mut rules = Vec::new();
86
87        // Load rules from configuration
88        for rule_config in &config.url_mapping_rules {
89            if let Ok(rule) = Self::create_rule_from_config(rule_config) {
90                rules.push(rule);
91            } else {
92                warn!("Failed to create rule from config: {}", rule_config.name);
93            }
94        }
95
96        // If no rules in config, use built-in rules as fallback
97        if rules.is_empty() {
98            warn!("No URL mapping rules found in config, using built-in rules");
99            rules.extend(Self::create_builtin_rules()?);
100        }
101
102        // Sort rules by priority
103        rules.sort_by_key(|rule| rule.priority);
104
105        info!(
106            "Initialized URL mapper with {} rules for region {:?}",
107            rules.len(),
108            region
109        );
110
111        // CDN quality assessor will be created separately to avoid async complexity
112        let quality_assessor = None;
113
114        Ok(Self {
115            rules,
116            current_region: region,
117            cache: Self::create_cache(),
118            cache_enabled: config.general.enable_url_cache,
119            cache_ttl: Duration::from_secs(config.general.url_cache_ttl),
120            max_cache_entries: config.general.max_cache_entries,
121            server_tracker: Arc::new(Mutex::new(ServerTracker::new())),
122            quality_assessor,
123        })
124    }
125
126    /// Create a rule from configuration
127    fn create_rule_from_config(config: &UrlMappingRuleConfig) -> Result<UrlMappingRule> {
128        let pattern = Regex::new(&config.pattern).map_err(|e| {
129            TurboCdnError::config(format!("Invalid regex pattern '{}': {}", config.pattern, e))
130        })?;
131
132        Ok(UrlMappingRule {
133            pattern,
134            replacements: config.replacements.clone(),
135            regions: config.regions.clone(),
136            priority: config.priority,
137            enabled: config.enabled,
138        })
139    }
140
141    /// Map a URL to the best CDN alternative with performance-based ordering
142    pub fn map_url(&mut self, original_url: &str) -> Result<Vec<String>> {
143        debug!("Mapping URL: {}", original_url);
144
145        // Check cache first
146        if self.cache_enabled {
147            if let Some(entry) = self.cache.get(original_url) {
148                if !entry.is_expired() {
149                    debug!("Cache hit for URL: {}", original_url);
150                    return Ok(entry.urls.clone());
151                }
152            }
153        }
154
155        let mut mapped_urls = Vec::new();
156
157        // Try to match against rules
158        for rule in &self.rules {
159            if !rule.enabled {
160                continue;
161            }
162
163            // Check if rule applies to current region
164            if !rule.regions.is_empty() && !rule.regions.contains(&self.current_region) {
165                continue;
166            }
167
168            // Try to match the pattern
169            if let Some(captures) = rule.pattern.captures(original_url) {
170                debug!(
171                    "Matched rule '{}' with pattern: {}",
172                    rule.pattern.as_str(),
173                    rule.pattern.as_str()
174                );
175
176                // Generate replacement URLs
177                for template in &rule.replacements {
178                    if let Ok(mapped_url) = Self::apply_template(template, &captures) {
179                        if mapped_url != original_url {
180                            mapped_urls.push(mapped_url);
181                        }
182                    }
183                }
184
185                // Only use the first matching rule
186                break;
187            }
188        }
189
190        // Always include the original URL as fallback
191        mapped_urls.push(original_url.to_string());
192
193        // Sort URLs by server performance if we have performance data
194        if mapped_urls.len() > 1 {
195            mapped_urls = self.sort_urls_by_performance(mapped_urls);
196        }
197
198        // Cache the result
199        if self.cache_enabled {
200            self.cache_result(original_url.to_string(), mapped_urls.clone());
201        }
202
203        debug!("Mapped to {} URLs: {:?}", mapped_urls.len(), mapped_urls);
204        Ok(mapped_urls)
205    }

207    /// Create built-in rules as fallback when no config rules are available
208    fn create_builtin_rules() -> Result<Vec<UrlMappingRule>> {
209        warn!("Using built-in fallback rules. Consider adding rules to your configuration file.");
210
211        // Create minimal fallback rules from embedded config
212        let fallback_configs = vec![UrlMappingRuleConfig {
213            name: "GitHub Releases - Global Fallback".to_string(),
214            pattern: r"^https://github\.com/([^/]+)/([^/]+)/releases/download/([^/]+)/(.+)$"
215                .to_string(),
216            replacements: vec![
217                "https://ghproxy.net/https://github.com/$1/$2/releases/download/$3/$4".to_string(),
218                "https://github.com/$1/$2/releases/download/$3/$4".to_string(),
219            ],
220            regions: vec![Region::Global],
221            priority: 100, // Lower priority than config rules
222            enabled: true,
223        }];
224
225        let mut rules = Vec::new();
226        for config in fallback_configs {
227            if let Ok(rule) = Self::create_rule_from_config(&config) {
228                rules.push(rule);
229            }
230        }
231
232        Ok(rules)
233    }
234
235    /// Apply template with regex captures
236    fn apply_template(template: &str, captures: &regex::Captures) -> Result<String> {
237        let mut result = template.to_string();
238
239        // Replace $1, $2, etc. with capture groups
240        for (i, capture) in captures.iter().enumerate() {
241            if let Some(matched) = capture {
242                let placeholder = format!("${}", i);
243                result = result.replace(&placeholder, matched.as_str());
244            }
245        }
246
247        Ok(result)
248    }

261    /// Sort URLs by server performance (best performing first)
262    /// Note: CDN quality assessment is handled separately to avoid async complexity
263    fn sort_urls_by_performance(&self, mut urls: Vec<String>) -> Vec<String> {
264        // Use server tracker performance for sorting
265        if let Ok(tracker) = self.server_tracker.lock() {
266            urls.sort_by(|a, b| {
267                let score_a = tracker
268                    .get_stats(a)
269                    .map(|s| s.performance_score())
270                    .unwrap_or(0.0);
271                let score_b = tracker
272                    .get_stats(b)
273                    .map(|s| s.performance_score())
274                    .unwrap_or(0.0);
275                // Higher score = better performance, so reverse order
276                score_b
277                    .partial_cmp(&score_a)
278                    .unwrap_or(std::cmp::Ordering::Equal)
279            });
280        }
281        urls
282    }

289    /// Cache the mapping result
290    fn cache_result(&self, url: String, urls: Vec<String>) {
291        // Clean up expired entries if cache is getting full
292        if self.cache.len() >= self.max_cache_entries {
293            self.cleanup_cache();
294        }
295
296        let entry = CacheEntry::new(urls, self.cache_ttl);
297        self.cache.insert(url, entry);
298    }
299
300    /// Clean up expired cache entries
301    fn cleanup_cache(&self) {
302        // Remove expired entries
303        self.cache.retain(|_, entry| !entry.is_expired());
304
305        // If still too many entries, remove oldest ones
306        if self.cache.len() >= self.max_cache_entries {
307            let mut entries: Vec<_> = self
308                .cache
309                .iter()
310                .map(|item| (item.key().clone(), item.value().cached_at))
311                .collect();
312            entries.sort_by_key(|(_, cached_at)| *cached_at);
313
314            let to_remove = entries.len() - self.max_cache_entries / 2;
315            let urls_to_remove: Vec<_> = entries
316                .into_iter()
317                .take(to_remove)
318                .map(|(url, _)| url)
319                .collect();
320
321            for url in urls_to_remove {
322                self.cache.remove(&url);
323            }
324        }
325    }

327    /// Get server tracker for external use
328    pub fn get_server_tracker(&self) -> Arc<Mutex<ServerTracker>> {
329        self.server_tracker.clone()
330    }
331
332    /// Get number of active rules
333    pub fn rule_count(&self) -> usize {
334        self.rules.iter().filter(|rule| rule.enabled).count()
335    }
336
337    /// Get cache statistics
338    pub fn cache_stats(&self) -> (usize, usize) {
339        let total = self.cache.len();
340        let expired = self
341            .cache
342            .iter()
343            .filter(|item| item.value().is_expired())
344            .count();
345        (total, expired)
346    }

353    #[test]
354    fn test_github_url_mapping() {
355        let config = TurboCdnConfig::default();
356        let mut mapper = UrlMapper::new(&config, Region::China).unwrap();
357
358        let original_url = "https://github.com/BurntSushi/ripgrep/releases/download/14.1.1/ripgrep-14.1.1-x86_64-pc-windows-msvc.zip";
359        let mapped_urls = mapper.map_url(original_url).unwrap();
360
361        assert!(!mapped_urls.is_empty());
362        assert!(mapped_urls.contains(&original_url.to_string()));
363    }
