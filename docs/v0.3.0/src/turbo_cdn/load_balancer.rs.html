4//! Intelligent Load Balancer
5//!
6//! This module implements smart load balancing strategies with server load monitoring,
7//! request distribution algorithms, and automatic failover mechanisms.
8
9use dashmap::DashMap;
10use std::sync::atomic::{AtomicU64, Ordering};
11use std::sync::Arc;
12use std::time::{Duration, Instant};
13use tokio::sync::RwLock;
14use tracing::{debug, warn};
15
16/// Server load metrics
17#[derive(Debug, Clone)]
18pub struct ServerLoad {
19    /// Current active connections
20    pub active_connections: u32,
21    /// Average response time in milliseconds
22    pub avg_response_time: f64,
23    /// Success rate (0.0 to 1.0)
24    pub success_rate: f64,
25    /// Current bandwidth utilization (bytes/sec)
26    pub bandwidth_usage: f64,
27    /// Server health score (0.0 to 1.0)
28    pub health_score: f64,
29    /// Last update timestamp
30    pub last_updated: Instant,
31    /// Total requests handled
32    pub total_requests: u64,
33    /// Failed requests
34    pub failed_requests: u64,
35}
36
37impl Default for ServerLoad {
38    fn default() -> Self {
39        Self {
40            active_connections: 0,
41            avg_response_time: 100.0,
42            success_rate: 1.0,
43            bandwidth_usage: 0.0,
44            health_score: 1.0,
45            last_updated: Instant::now(),
46            total_requests: 0,
47            failed_requests: 0,
48        }
49    }
50}

52/// Load balancing strategy
53#[derive(Debug, Clone, PartialEq, Copy)]
54pub enum LoadBalancingStrategy {
55    /// Round-robin distribution
56    RoundRobin,
57    /// Weighted round-robin based on server capacity
58    WeightedRoundRobin,
59    /// Least connections first
60    LeastConnections,
61    /// Fastest response time first
62    FastestResponse,
63    /// Health-based selection
64    HealthBased,
65    /// Adaptive strategy that changes based on conditions
66    Adaptive,
67}
68
69/// Intelligent load balancer
70#[derive(Debug)]
71pub struct LoadBalancer {
72    /// Server load metrics
73    server_loads: Arc<DashMap<String, ServerLoad>>,
74    /// Load balancing strategy
75    strategy: Arc<RwLock<LoadBalancingStrategy>>,
76    /// Round-robin counter
77    round_robin_counter: AtomicU64,
78    /// Failover threshold
79    failover_threshold: f64,
80    /// Health check interval
81    health_check_interval: Duration,
82    /// Last health check time
83    last_health_check: Arc<RwLock<Instant>>,
84}
85
86impl LoadBalancer {
87    /// Create a new load balancer
88    pub fn new(strategy: LoadBalancingStrategy) -> Self {
89        Self {
90            server_loads: Arc::new(DashMap::new()),
91            strategy: Arc::new(RwLock::new(strategy)),
92            round_robin_counter: AtomicU64::new(0),
93            failover_threshold: 0.3, // 30% health threshold
94            health_check_interval: Duration::from_secs(30),
95            last_health_check: Arc::new(RwLock::new(Instant::now())),
96        }
97    }
98
99    /// Select the best server from available options
100    pub async fn select_server(&self, servers: &[String]) -> Option<String> {
101        if servers.is_empty() {
102            return None;
103        }
104
105        // Ensure all servers are tracked
106        for server in servers {
107            if !self.server_loads.contains_key(server) {
108                self.server_loads
109                    .insert(server.clone(), ServerLoad::default());
110            }
111        }
112
113        // Maybe perform health check
114        self.maybe_health_check(servers).await;
115
116        // Filter healthy servers
117        let healthy_servers: Vec<String> = servers
118            .iter()
119            .filter(|server| {
120                self.server_loads
121                    .get(*server)
122                    .map(|load| load.health_score >= self.failover_threshold)
123                    .unwrap_or(true)
124            })
125            .cloned()
126            .collect();
127
128        if healthy_servers.is_empty() {
129            warn!("No healthy servers available, using all servers");
130            return self.select_by_strategy(servers).await;
131        }
132
133        self.select_by_strategy(&healthy_servers).await
134    }
135
136    /// Select server based on current strategy
137    async fn select_by_strategy(&self, servers: &[String]) -> Option<String> {
138        let strategy = *self.strategy.read().await;
139
140        match strategy {
141            LoadBalancingStrategy::RoundRobin => self.round_robin_select(servers),
142            LoadBalancingStrategy::WeightedRoundRobin => {
143                self.weighted_round_robin_select(servers).await
144            }
145            LoadBalancingStrategy::LeastConnections => self.least_connections_select(servers),
146            LoadBalancingStrategy::FastestResponse => self.fastest_response_select(servers),
147            LoadBalancingStrategy::HealthBased => self.health_based_select(servers),
148            LoadBalancingStrategy::Adaptive => {
149                // Use health-based selection for adaptive strategy to avoid recursion
150                self.health_based_select(servers)
151            }
152        }
153    }
154
155    /// Round-robin server selection
156    fn round_robin_select(&self, servers: &[String]) -> Option<String> {
157        if servers.is_empty() {
158            return None;
159        }
160
161        let index =
162            self.round_robin_counter.fetch_add(1, Ordering::Relaxed) as usize % servers.len();
163        Some(servers[index].clone())
164    }
165
166    /// Weighted round-robin based on server health
167    async fn weighted_round_robin_select(&self, servers: &[String]) -> Option<String> {
168        if servers.is_empty() {
169            return None;
170        }
171
172        // Calculate weights based on health scores
173        let mut weighted_servers = Vec::new();
174        for server in servers {
175            if let Some(load) = self.server_loads.get(server) {
176                let weight = (load.health_score * 10.0) as usize + 1; // 1-11 weight range
177                for _ in 0..weight {
178                    weighted_servers.push(server.clone());
179                }
180            } else {
181                weighted_servers.push(server.clone()); // Default weight of 1
182            }
183        }
184
185        if weighted_servers.is_empty() {
186            return servers.first().cloned();
187        }
188
189        let index = self.round_robin_counter.fetch_add(1, Ordering::Relaxed) as usize
190            % weighted_servers.len();
191        Some(weighted_servers[index].clone())
192    }
193
194    /// Select server with least connections
195    fn least_connections_select(&self, servers: &[String]) -> Option<String> {
196        servers
197            .iter()
198            .min_by_key(|server| {
199                self.server_loads
200                    .get(*server)
201                    .map(|load| load.active_connections)
202                    .unwrap_or(0)
203            })
204            .cloned()
205    }
206
207    /// Select server with fastest response time
208    fn fastest_response_select(&self, servers: &[String]) -> Option<String> {
209        servers
210            .iter()
211            .min_by(|a, b| {
212                let time_a = self
213                    .server_loads
214                    .get(*a)
215                    .map(|load| load.avg_response_time)
216                    .unwrap_or(1000.0);
217                let time_b = self
218                    .server_loads
219                    .get(*b)
220                    .map(|load| load.avg_response_time)
221                    .unwrap_or(1000.0);
222                time_a
223                    .partial_cmp(&time_b)
224                    .unwrap_or(std::cmp::Ordering::Equal)
225            })
226            .cloned()
227    }
228
229    /// Select server based on health score
230    fn health_based_select(&self, servers: &[String]) -> Option<String> {
231        servers
232            .iter()
233            .max_by(|a, b| {
234                let health_a = self
235                    .server_loads
236                    .get(*a)
237                    .map(|load| load.health_score)
238                    .unwrap_or(0.5);
239                let health_b = self
240                    .server_loads
241                    .get(*b)
242                    .map(|load| load.health_score)
243                    .unwrap_or(0.5);
244                health_a
245                    .partial_cmp(&health_b)
246                    .unwrap_or(std::cmp::Ordering::Equal)
247            })
248            .cloned()
249    }
250
251    /// Adaptive selection that changes strategy based on conditions
252    #[allow(dead_code)]
253    async fn adaptive_select(&self, servers: &[String]) -> Option<String> {
254        // Analyze current conditions and choose best strategy
255        let avg_health = self.calculate_average_health(servers);
256        let load_variance = self.calculate_load_variance(servers);
257
258        let optimal_strategy = if avg_health < 0.7 {
259            LoadBalancingStrategy::HealthBased
260        } else if load_variance > 0.3 {
261            LoadBalancingStrategy::LeastConnections
262        } else {
263            LoadBalancingStrategy::FastestResponse
264        };
265
266        // Temporarily use optimal strategy
267        let original_strategy = *self.strategy.read().await;
268        *self.strategy.write().await = optimal_strategy;
269
270        let result = self.select_by_strategy(servers).await;
271
272        // Restore original strategy
273        *self.strategy.write().await = original_strategy;
274
275        result
276    }

278    /// Record request start (increment active connections)
279    pub fn record_request_start(&self, server: &str) {
280        if let Some(mut load) = self.server_loads.get_mut(server) {
281            load.active_connections += 1;
282            load.total_requests += 1;
283        }
284    }
285
286    /// Record request completion
287    pub fn record_request_completion(&self, server: &str, duration: Duration, success: bool) {
288        if let Some(mut load) = self.server_loads.get_mut(server) {
289            load.active_connections = load.active_connections.saturating_sub(1);
290
291            // Update response time with exponential moving average
292            let response_time = duration.as_millis() as f64;
293            load.avg_response_time = load.avg_response_time * 0.8 + response_time * 0.2;
294
295            // Update success rate
296            if !success {
297                load.failed_requests += 1;
298            }
299            load.success_rate = if load.total_requests > 0 {
300                (load.total_requests - load.failed_requests) as f64 / load.total_requests as f64
301            } else {
302                1.0
303            };
304
305            // Update health score
306            load.health_score = self.calculate_health_score(&load);
307            load.last_updated = Instant::now();
308        }
309    }
310
311    /// Calculate server health score
312    fn calculate_health_score(&self, load: &ServerLoad) -> f64 {
313        // Weighted combination of metrics
314        let response_score = (500.0 - load.avg_response_time.min(500.0)) / 500.0;
315        let success_score = load.success_rate;
316        let connection_score = if load.active_connections < 10 {
317            1.0
318        } else {
319            10.0 / load.active_connections as f64
320        };
321
322        // Weighted average: response 30%, success 50%, connections 20%
323        (response_score * 0.3 + success_score * 0.5 + connection_score * 0.2).clamp(0.0, 1.0)
324    }
325
326    /// Calculate average health across servers
327    #[allow(dead_code)]
328    fn calculate_average_health(&self, servers: &[String]) -> f64 {
329        if servers.is_empty() {
330            return 0.0;
331        }
332
333        let total_health: f64 = servers
334            .iter()
335            .map(|server| {
336                self.server_loads
337                    .get(server)
338                    .map(|load| load.health_score)
339                    .unwrap_or(0.5)
340            })
341            .sum();
342
343        total_health / servers.len() as f64
344    }
345
346    /// Calculate load variance across servers
347    #[allow(dead_code)]
348    fn calculate_load_variance(&self, servers: &[String]) -> f64 {
349        if servers.len() < 2 {
350            return 0.0;
351        }
352
353        let loads: Vec<f64> = servers
354            .iter()
355            .map(|server| {
356                self.server_loads
357                    .get(server)
358                    .map(|load| load.active_connections as f64)
359                    .unwrap_or(0.0)
360            })
361            .collect();
362
363        let mean = loads.iter().sum::<f64>() / loads.len() as f64;
364        let variance =
365            loads.iter().map(|load| (load - mean).powi(2)).sum::<f64>() / loads.len() as f64;
366
367        variance.sqrt() / (mean + 1.0) // Normalized variance
368    }

370    /// Maybe perform health check
371    async fn maybe_health_check(&self, servers: &[String]) {
372        let last_check = *self.last_health_check.read().await;
373        if last_check.elapsed() >= self.health_check_interval {
374            self.perform_health_check(servers).await;
375            *self.last_health_check.write().await = Instant::now();
376        }
377    }
378
379    /// Perform health check on servers
380    async fn perform_health_check(&self, servers: &[String]) {
381        for server in servers {
382            if let Some(mut load) = self.server_loads.get_mut(server) {
383                // Check if server data is stale
384                if load.last_updated.elapsed() > Duration::from_secs(300) {
385                    // Gradually decrease health score for stale servers
386                    load.health_score *= 0.9;
387                    debug!(
388                        "Decreased health score for stale server {}: {:.2}",
389                        server, load.health_score
390                    );
391                }
392            }
393        }
394    }

396    /// Get load balancer statistics
397    pub fn get_stats(&self) -> LoadBalancerStats {
398        let total_servers = self.server_loads.len();
399        let healthy_servers = self
400            .server_loads
401            .iter()
402            .filter(|entry| entry.health_score >= self.failover_threshold)
403            .count();
404
405        let total_connections: u32 = self
406            .server_loads
407            .iter()
408            .map(|entry| entry.active_connections)
409            .sum();
410
411        let avg_health = if total_servers > 0 {
412            self.server_loads
413                .iter()
414                .map(|entry| entry.health_score)
415                .sum::<f64>()
416                / total_servers as f64
417        } else {
418            0.0
419        };
420
421        LoadBalancerStats {
422            total_servers,
423            healthy_servers,
424            total_connections,
425            avg_health,
426            strategy: format!(
427                "{:?}",
428                self.strategy
429                    .try_read()
430                    .map(|s| *s)
431                    .unwrap_or(LoadBalancingStrategy::RoundRobin)
432            ),
433        }
434    }

439pub struct LoadBalancerStats {

443    pub avg_health: f64,
444    pub strategy: String,
445}

447impl std::fmt::Display for LoadBalancerStats {

454            self.avg_health * 100.0,
455            self.strategy
456        )
