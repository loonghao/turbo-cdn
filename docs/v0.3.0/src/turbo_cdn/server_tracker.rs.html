4//! Server performance tracking and intelligent selection
5//!
6//! This module provides server performance tracking similar to aria2's feedback
7//! URI selector, allowing the downloader to learn from historical performance
8//! and select the fastest servers automatically.
9
10use std::collections::HashMap;
11use std::time::{Duration, Instant};
12use tracing::{debug, info, warn};
13
14/// Server performance statistics
15#[derive(Debug, Clone)]
16pub struct ServerStats {
17    /// Average download speed in bytes per second
18    pub average_speed: f64,
19    /// Success rate (0.0 to 1.0)
20    pub success_rate: f64,
21    /// Average response time
22    pub average_response_time: Duration,
23    /// Total number of attempts
24    pub total_attempts: u32,
25    /// Number of successful downloads
26    pub successful_downloads: u32,
27    /// Number of failed downloads
28    pub failed_downloads: u32,
29    /// Last update timestamp
30    pub last_updated: Instant,
31    /// Recent speed samples (for calculating moving average)
32    pub recent_speeds: Vec<f64>,
33    /// Recent response times
34    pub recent_response_times: Vec<Duration>,
35}
36
37impl Default for ServerStats {
38    fn default() -> Self {
39        Self {
40            average_speed: 0.0,
41            success_rate: 1.0, // Start optimistic
42            average_response_time: Duration::from_millis(100),
43            total_attempts: 0,
44            successful_downloads: 0,
45            failed_downloads: 0,
46            last_updated: Instant::now(),
47            recent_speeds: Vec::new(),
48            recent_response_times: Vec::new(),
49        }
50    }
51}
52
53impl ServerStats {
54    /// Calculate performance score (0.0 to 1.0, higher is better)
55    pub fn performance_score(&self) -> f64 {
56        if self.total_attempts == 0 {
57            return 0.5; // Neutral score for untested servers
58        }
59
60        // Weight factors
61        let speed_weight = 0.4;
62        let success_weight = 0.4;
63        let latency_weight = 0.2;
64
65        // Normalize speed (assume 10MB/s is excellent)
66        let speed_score = (self.average_speed / (10.0 * 1024.0 * 1024.0)).min(1.0);
67
68        // Success rate is already normalized
69        let success_score = self.success_rate;
70
71        // Normalize latency (assume 50ms is excellent, 1000ms is poor)
72        let latency_ms = self.average_response_time.as_millis() as f64;
73        let latency_score = (1000.0 - latency_ms.min(1000.0)) / 1000.0;
74
75        speed_weight * speed_score + success_weight * success_score + latency_weight * latency_score
76    }
77
78    /// Update statistics with new download result
79    pub fn update_with_result(&mut self, speed: f64, response_time: Duration, success: bool) {
80        self.total_attempts += 1;
81        self.last_updated = Instant::now();
82
83        if success {
84            self.successful_downloads += 1;
85
86            // Update speed statistics
87            self.recent_speeds.push(speed);
88            if self.recent_speeds.len() > 10 {
89                self.recent_speeds.remove(0);
90            }
91            self.average_speed =
92                self.recent_speeds.iter().sum::<f64>() / self.recent_speeds.len() as f64;
93        } else {
94            self.failed_downloads += 1;
95        }
96
97        // Update response time statistics
98        self.recent_response_times.push(response_time);
99        if self.recent_response_times.len() > 10 {
100            self.recent_response_times.remove(0);
101        }
102        self.average_response_time = Duration::from_millis(
103            (self
104                .recent_response_times
105                .iter()
106                .map(|d| d.as_millis())
107                .sum::<u128>()
108                / self.recent_response_times.len() as u128) as u64,
109        );
110
111        // Update success rate
112        self.success_rate = self.successful_downloads as f64 / self.total_attempts as f64;
113
114        debug!(
115            "Updated server stats: speed={:.2} MB/s, success_rate={:.2}, response_time={:?}, score={:.3}",
116            self.average_speed / 1024.0 / 1024.0,
117            self.success_rate,
118            self.average_response_time,
119            self.performance_score()
120        );
121    }
122}

124/// Server performance tracker with intelligent selection
125#[derive(Debug)]
126pub struct ServerTracker {
127    server_stats: HashMap<String, ServerStats>,
128    max_servers_to_track: usize,
129}
130
131impl ServerTracker {
132    /// Create a new server tracker
133    pub fn new() -> Self {
134        Self {
135            server_stats: HashMap::new(),
136            max_servers_to_track: 100, // Limit memory usage
137        }
138    }
139
140    /// Select the best servers from a list of URLs
141    pub fn select_best_servers(&self, urls: &[String], max_count: usize) -> Vec<String> {
142        if urls.is_empty() {
143            return Vec::new();
144        }
145
146        let default_stats = ServerStats::default();
147        let mut scored_urls: Vec<(String, f64)> = urls
148            .iter()
149            .map(|url| {
150                let stats = self.server_stats.get(url).unwrap_or(&default_stats);
151                let score = stats.performance_score();
152                (url.clone(), score)
153            })
154            .collect();
155
156        // Sort by score (descending)
157        scored_urls.sort_by(|a, b| b.1.partial_cmp(&a.1).unwrap_or(std::cmp::Ordering::Equal));
158
159        let selected: Vec<String> = scored_urls
160            .into_iter()
161            .take(max_count)
162            .map(|(url, score)| {
163                debug!("Selected server: {} (score: {:.3})", url, score);
164                url
165            })
166            .collect();
167
168        info!(
169            "Selected {} servers from {} candidates",
170            selected.len(),
171            urls.len()
172        );
173        selected
174    }
175
176    /// Record a successful download
177    pub fn record_success(&mut self, url: &str, speed: f64, response_time: Duration) {
178        self.ensure_capacity();
179        let stats = self.server_stats.entry(url.to_string()).or_default();
180        stats.update_with_result(speed, response_time, true);
181    }
182
183    /// Record a failed download
184    pub fn record_failure(&mut self, url: &str, response_time: Duration) {
185        self.ensure_capacity();
186        let stats = self.server_stats.entry(url.to_string()).or_default();
187        stats.update_with_result(0.0, response_time, false);
188    }
189
190    /// Get statistics for a specific server
191    pub fn get_stats(&self, url: &str) -> Option<&ServerStats> {
192        self.server_stats.get(url)
193    }
194
195    /// Get all server statistics
196    pub fn get_all_stats(&self) -> &HashMap<String, ServerStats> {
197        &self.server_stats
198    }
199
200    /// Clear old statistics to prevent memory bloat
201    fn ensure_capacity(&mut self) {
202        if self.server_stats.len() >= self.max_servers_to_track {
203            // Collect URLs to remove (oldest entries by last_updated)
204            let mut entries: Vec<_> = self
205                .server_stats
206                .iter()
207                .map(|(url, stats)| (url.clone(), stats.last_updated))
208                .collect();
209            entries.sort_by_key(|(_, last_updated)| *last_updated);
210
211            let to_remove = entries.len() - self.max_servers_to_track + 10; // Remove 10 extra
212            let urls_to_remove: Vec<String> = entries
213                .into_iter()
214                .take(to_remove)
215                .map(|(url, _)| url)
216                .collect();
217
218            for url in urls_to_remove {
219                self.server_stats.remove(&url);
220            }
221
222            warn!(
223                "Cleaned up old server statistics, now tracking {} servers",
224                self.server_stats.len()
225            );
226        }
227    }
228
229    /// Get performance summary
230    pub fn get_performance_summary(&self) -> PerformanceSummary {
231        if self.server_stats.is_empty() {
232            return PerformanceSummary::default();
233        }
234
235        let total_servers = self.server_stats.len();
236        let total_attempts: u32 = self.server_stats.values().map(|s| s.total_attempts).sum();
237        let total_successes: u32 = self
238            .server_stats
239            .values()
240            .map(|s| s.successful_downloads)
241            .sum();
242        let average_speed: f64 = self
243            .server_stats
244            .values()
245            .filter(|s| s.average_speed > 0.0)
246            .map(|s| s.average_speed)
247            .sum::<f64>()
248            / self
249                .server_stats
250                .values()
251                .filter(|s| s.average_speed > 0.0)
252                .count()
253                .max(1) as f64;
254
255        let best_server = self
256            .server_stats
257            .iter()
258            .max_by(|(_, a), (_, b)| {
259                a.performance_score()
260                    .partial_cmp(&b.performance_score())
261                    .unwrap_or(std::cmp::Ordering::Equal)
262            })
263            .map(|(url, stats)| (url.clone(), stats.performance_score()));
264
265        PerformanceSummary {
266            total_servers,
267            total_attempts,
268            total_successes,
269            overall_success_rate: if total_attempts > 0 {
270                total_successes as f64 / total_attempts as f64
271            } else {
272                0.0
273            },
274            average_speed,
275            best_server,
276        }
277    }
278}

280impl Default for ServerTracker {
281    fn default() -> Self {
282        Self::new()
283    }

286/// Performance summary for reporting
287#[derive(Debug, Clone)]
288pub struct PerformanceSummary {
289    pub total_servers: usize,
290    pub total_attempts: u32,
291    pub total_successes: u32,
292    pub overall_success_rate: f64,
293    pub average_speed: f64,
294    pub best_server: Option<(String, f64)>,
295}
296
297impl Default for PerformanceSummary {
298    fn default() -> Self {
299        Self {
300            total_servers: 0,

306        }

314    #[test]
315    fn test_server_stats_performance_score() {
316        let mut stats = ServerStats::default();
317
318        // Test with good performance
319        stats.update_with_result(5.0 * 1024.0 * 1024.0, Duration::from_millis(50), true);
320        assert!(stats.performance_score() > 0.5);
321
322        // Test with poor performance
323        stats.update_with_result(100.0 * 1024.0, Duration::from_millis(500), false);
324        assert!(stats.performance_score() < 0.8);
325    }
326
327    #[test]
328    fn test_server_tracker_selection() {
329        let mut tracker = ServerTracker::new();
330
331        // Record some performance data
332        tracker.record_success(
333            "http://fast.example.com",
334            10.0 * 1024.0 * 1024.0,
335            Duration::from_millis(50),
336        );
337        tracker.record_success(
338            "http://slow.example.com",
339            1.0 * 1024.0 * 1024.0,
340            Duration::from_millis(200),
341        );
342        tracker.record_failure("http://bad.example.com", Duration::from_millis(1000));
343
344        let urls = vec![
345            "http://fast.example.com".to_string(),
346            "http://slow.example.com".to_string(),
347            "http://bad.example.com".to_string(),
348        ];
349
350        let selected = tracker.select_best_servers(&urls, 2);
351        assert_eq!(selected.len(), 2);
352        assert_eq!(selected[0], "http://fast.example.com");
353    }
