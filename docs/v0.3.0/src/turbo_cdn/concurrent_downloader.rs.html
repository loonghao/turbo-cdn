52/// High-performance concurrent downloader with dynamic segmentation
53///
54/// Provides intelligent chunked downloads with adaptive sizing, resume support,
55/// and server performance tracking for optimal download speeds.
56#[derive(Debug)]
57pub struct ConcurrentDownloader {
58    http_client: Client,
59    max_concurrent_chunks: usize,
60    initial_chunk_size: u64,
61    min_chunk_size: u64,
62    max_chunk_size: u64,
63    #[allow(dead_code)]
64    request_timeout: Duration,
65    adaptive_chunking_enabled: bool,
66    speed_threshold_bytes_per_sec: u64,
67    server_performance_tracker: std::sync::Arc<std::sync::Mutex<ServerTracker>>,
68}

77    /// Create a new concurrent downloader with configuration
78    pub fn with_config(config: &crate::config::TurboCdnConfig) -> Result<Self> {
79        let mut client_builder = Client::builder()
80            .timeout(Duration::from_secs(config.performance.timeout))
81            .user_agent(&config.general.user_agent)
82            .pool_max_idle_per_host(config.performance.pool_max_idle_per_host)
83            .pool_idle_timeout(Duration::from_secs(config.performance.pool_idle_timeout))
84            .tcp_keepalive(Duration::from_secs(config.performance.tcp_keepalive))
85            .tcp_nodelay(true) // Disable Nagle's algorithm for lower latency
86            .connection_verbose(config.general.debug) // Enable connection debugging if debug mode
87            .redirect(reqwest::redirect::Policy::limited(10)); // Allow up to 10 redirects
88
89        if config.performance.http2_prior_knowledge {
90            client_builder = client_builder.http2_prior_knowledge();
91        }
92
93        // Configure TLS settings for better performance
94        client_builder = client_builder
95            .http2_adaptive_window(true)
96            .http2_max_frame_size(Some(16384)); // 16KB frame size for better throughput
97
98        let client = client_builder
99            .build()
100            .map_err(|e| TurboCdnError::network(format!("Failed to create HTTP client: {}", e)))?;
101
102        Ok(Self {
103            http_client: client,
104            max_concurrent_chunks: config.performance.max_concurrent_downloads,
105            initial_chunk_size: config.performance.chunk_size,
106            min_chunk_size: config.performance.min_chunk_size,
107            max_chunk_size: config.performance.max_chunk_size,
108            request_timeout: Duration::from_secs(config.performance.timeout),
109            adaptive_chunking_enabled: config.performance.adaptive_chunking,
110            speed_threshold_bytes_per_sec: config.performance.speed_threshold_bytes_per_sec,
111            server_performance_tracker: std::sync::Arc::new(std::sync::Mutex::new(
112                ServerTracker::new(),
113            )),
114        })
115    }

117    /// Download a file from URL with automatic optimization and retry logic
118    pub async fn download<P: AsRef<Path>>(
119        &self,
120        urls: &[String],
121        output_path: P,
122        progress_tracker: Option<Arc<ProgressTracker>>,
123    ) -> Result<DownloadResult> {
124        let output_path = output_path.as_ref();
125        let start_time = Instant::now();
126
127        // Use intelligent server selection
128        let selected_urls = {
129            let tracker = self.server_performance_tracker.lock().unwrap();
130            tracker.select_best_servers(urls, self.max_concurrent_chunks.min(urls.len()))
131        };
132
133        info!(
134            "Selected {} URLs for download from {} candidates",
135            selected_urls.len(),
136            urls.len()
137        );
138
139        // Try each URL with retry logic
140        for (index, url) in selected_urls.iter().enumerate() {
141            debug!("Trying URL {}/{}: {}", index + 1, selected_urls.len(), url);
142
143            // Retry logic for each URL
144            for retry_attempt in 0..=3 {
145                // Max 3 retries per URL
146                let url_start_time = Instant::now();
147
148                if retry_attempt > 0 {
149                    debug!("Retry attempt {} for URL: {}", retry_attempt, url);
150                    // Exponential backoff: 1s, 2s, 4s
151                    let delay = Duration::from_secs(2_u64.pow(retry_attempt as u32 - 1));
152                    tokio::time::sleep(delay).await;
153                }
154
155                match self
156                    .download_single_url(url, output_path, progress_tracker.clone())
157                    .await
158                {
159                    Ok(mut result) => {
160                        let url_duration = url_start_time.elapsed();
161                        result.duration = start_time.elapsed();
162                        result.speed = result.size as f64 / result.duration.as_secs_f64();
163
164                        // Record successful download
165                        {
166                            let mut tracker = self.server_performance_tracker.lock().unwrap();
167                            tracker.record_success(url, result.speed, url_duration);
168                        }
169
170                        info!(
171                            "Download completed: {} bytes in {:.2}s ({:.2} MB/s) from {} (attempt {})",
172                            result.size,
173                            result.duration.as_secs_f64(),
174                            result.speed / 1024.0 / 1024.0,
175                            url,
176                            retry_attempt + 1
177                        );
178                        return Ok(result);
179                    }
180                    Err(e) => {
181                        let url_duration = url_start_time.elapsed();
182
183                        // Record failed download
184                        {
185                            let mut tracker = self.server_performance_tracker.lock().unwrap();
186                            tracker.record_failure(url, url_duration);
187                        }
188
189                        warn!("Attempt {} failed for {}: {}", retry_attempt + 1, url, e);
190
191                        // If this was the last retry for this URL, try next URL
192                        if retry_attempt == 3 {
193                            break;
194                        }
195                    }
196                }
197            }
198
199            // If we've exhausted all retries for all URLs, return error
200            if index == selected_urls.len() - 1 {
201                return Err(TurboCdnError::network(
202                    "All download URLs failed after retries".to_string(),
203                ));
204            }
205        }
206
207        Err(TurboCdnError::network(
208            "All download URLs failed".to_string(),
209        ))
210    }

212    /// Download from a single URL
213    async fn download_single_url<P: AsRef<Path>>(
214        &self,
215        url: &str,
216        output_path: P,
217        progress_tracker: Option<Arc<ProgressTracker>>,
218    ) -> Result<DownloadResult> {
219        let output_path = output_path.as_ref();
220
221        // Check if file already exists and get its size
222        let existing_size = if output_path.exists() {
223            tokio::fs::metadata(output_path)
224                .await
225                .map(|m| m.len())
226                .unwrap_or(0)
227        } else {
228            0
229        };
230
231        // Get file info from server
232        let file_info = self.get_file_info(url).await?;
233
234        // Check if file is already complete
235        if existing_size == file_info.total_size {
236            info!(
237                "File already exists and is complete: {}",
238                output_path.display()
239            );
240            return Ok(DownloadResult {
241                path: output_path.to_path_buf(),
242                size: existing_size,
243                duration: Duration::from_secs(0),
244                speed: 0.0,
245                url: url.to_string(),
246                resumed: false,
247            });
248        }
249
250        // Determine download strategy
251        if file_info.supports_ranges && file_info.total_size > self.min_chunk_size * 2 {
252            // Use concurrent chunked download
253            self.download_with_chunks(
254                url,
255                output_path,
256                &file_info,
257                existing_size,
258                progress_tracker,
259            )
260            .await
261        } else {
262            // Use single-threaded download
263            self.download_single_thread(url, output_path, existing_size, progress_tracker)
264                .await
265        }
266    }

268    /// Get file information from server
269    async fn get_file_info(&self, url: &str) -> Result<FileInfo> {
270        debug!("Getting file info for: {}", url);
271
272        let response = self
273            .http_client
274            .head(url)
275            .send()
276            .await
277            .map_err(|e| TurboCdnError::network(format!("Failed to get file info: {}", e)))?;
278
279        if !response.status().is_success() {
280            return Err(TurboCdnError::network(format!(
281                "Server returned error: {}",
282                response.status()
283            )));
284        }
285
286        let total_size = response
287            .headers()
288            .get("content-length")
289            .and_then(|v| v.to_str().ok())
290            .and_then(|s| s.parse::<u64>().ok())
291            .unwrap_or(0);
292
293        let supports_ranges = response
294            .headers()
295            .get("accept-ranges")
296            .map(|v| v.to_str().unwrap_or("").contains("bytes"))
297            .unwrap_or(false);
298
299        debug!(
300            "File info: size={}, supports_ranges={}",
301            total_size, supports_ranges
302        );
303
304        Ok(FileInfo {
305            total_size,
306            supports_ranges,
307        })
308    }

310    /// Download with concurrent chunks
311    async fn download_with_chunks<P: AsRef<Path>>(
312        &self,
313        url: &str,
314        output_path: P,
315        file_info: &FileInfo,
316        existing_size: u64,
317        progress_tracker: Option<Arc<ProgressTracker>>,
318    ) -> Result<DownloadResult> {
319        let output_path = output_path.as_ref();
320        let remaining_size = file_info.total_size - existing_size;
321
322        info!(
323            "Starting chunked download: {} bytes remaining, {} chunks",
324            remaining_size, self.max_concurrent_chunks
325        );
326
327        // Calculate chunks
328        let chunks = self.calculate_chunks(existing_size, file_info.total_size);
329        debug!("Created {} chunks", chunks.len());
330
331        // Create or open output file
332        let file = if existing_size > 0 {
333            OpenOptions::new()
334                .write(true)
335                .open(output_path)
336                .await
337                .map_err(|e| TurboCdnError::io(format!("Failed to open file: {}", e)))?
338        } else {
339            File::create(output_path)
340                .await
341                .map_err(|e| TurboCdnError::io(format!("Failed to create file: {}", e)))?
342        };
343
344        let file = Arc::new(tokio::sync::Mutex::new(file));
345
346        // Limit concurrent downloads
347        let semaphore = Arc::new(Semaphore::new(self.max_concurrent_chunks));
348
349        // Download chunks concurrently
350        let mut tasks = Vec::new();
351        for chunk in chunks {
352            let client = self.http_client.clone();
353            let url = url.to_string();
354            let file = file.clone();
355            let semaphore = semaphore.clone();
356            let progress_tracker = progress_tracker.clone();
357
358            let task = tokio::spawn(async move {
359                let _permit = semaphore.acquire().await.unwrap();
360                Self::download_chunk(client, &url, chunk, file, progress_tracker).await
361            });
362
363            tasks.push(task);
364        }
365
366        // Wait for all chunks to complete
367        for task in tasks {
368            task.await
369                .map_err(|e| TurboCdnError::network(format!("Chunk download failed: {}", e)))??;
370        }
371
372        Ok(DownloadResult {
373            path: output_path.to_path_buf(),
374            size: file_info.total_size,
375            duration: Duration::from_secs(0), // Will be set by caller
376            speed: 0.0,                       // Will be set by caller
377            url: url.to_string(),
378            resumed: existing_size > 0,
379        })
380    }

382    /// Calculate optimal chunks for download with adaptive sizing
383    fn calculate_chunks(&self, start_offset: u64, total_size: u64) -> Vec<ChunkInfo> {
384        self.calculate_adaptive_chunks(start_offset, total_size, None)
385    }

387    /// Calculate adaptive chunks based on current download speed
388    fn calculate_adaptive_chunks(
389        &self,
390        start_offset: u64,
391        total_size: u64,
392        current_speed: Option<u64>,
393    ) -> Vec<ChunkInfo> {
394        let remaining_size = total_size - start_offset;
395
396        // Determine optimal chunk size based on speed and file size
397        let optimal_chunk_size = if self.adaptive_chunking_enabled {
398            self.calculate_optimal_chunk_size(remaining_size, current_speed)
399        } else {
400            self.initial_chunk_size
401        };
402
403        // Ensure chunk size is within bounds
404        let chunk_size = optimal_chunk_size
405            .max(self.min_chunk_size)
406            .min(self.max_chunk_size);
407
408        // Calculate number of chunks, but don't exceed max_concurrent_chunks
409        let ideal_chunk_count = (remaining_size / chunk_size).max(1);
410        let actual_chunk_count = ideal_chunk_count.min(self.max_concurrent_chunks as u64);
411        let adjusted_chunk_size = remaining_size / actual_chunk_count;
412
413        let mut chunks = Vec::new();
414        let mut current_start = start_offset;
415        let mut index = 0;
416
417        while current_start < total_size && index < actual_chunk_count {
418            let chunk_end = if index == actual_chunk_count - 1 {
419                // Last chunk gets any remaining bytes
420                total_size - 1
421            } else {
422                (current_start + adjusted_chunk_size - 1).min(total_size - 1)
423            };
424
425            chunks.push(ChunkInfo {
426                start: current_start,
427                end: chunk_end,
428                index: index as usize,
429            });
430
431            current_start = chunk_end + 1;
432            index += 1;
433        }
434
435        debug!(
436            "Calculated {} chunks with size ~{} KB each (adaptive: {})",
437            chunks.len(),
438            adjusted_chunk_size / 1024,
439            self.adaptive_chunking_enabled
440        );
441
442        chunks
443    }

445    /// Calculate optimal chunk size based on file size and current speed
446    fn calculate_optimal_chunk_size(&self, file_size: u64, current_speed: Option<u64>) -> u64 {
447        // Base chunk size on file size - 更激进的分块策略
448        let size_based_chunk = if file_size > 100 * 1024 * 1024 {
449            // Large files (>100MB): use much larger chunks
450            self.initial_chunk_size * 4
451        } else if file_size > 10 * 1024 * 1024 {
452            // Medium files (10-100MB): use larger chunks
453            self.initial_chunk_size * 2
454        } else if file_size > 1024 * 1024 {
455            // Small files (1-10MB): use normal chunks
456            self.initial_chunk_size
457        } else {
458            // Very small files (<1MB): use smaller chunks but not too small
459            (self.initial_chunk_size / 4).max(self.min_chunk_size)
460        };
461
462        // Adjust based on current download speed if available
463        if let Some(speed) = current_speed {
464            if speed > self.speed_threshold_bytes_per_sec * 4 {
465                // Very fast connection: use much larger chunks
466                size_based_chunk * 3
467            } else if speed > self.speed_threshold_bytes_per_sec * 2 {
468                // Fast connection: use larger chunks
469                size_based_chunk * 2
470            } else if speed < self.speed_threshold_bytes_per_sec / 4 {
471                // Very slow connection: use smaller chunks for better parallelism
472                size_based_chunk / 3
473            } else if speed < self.speed_threshold_bytes_per_sec / 2 {
474                // Slow connection: use smaller chunks
475                size_based_chunk / 2
476            } else {
477                // Normal speed: use size-based chunk
478                size_based_chunk
479            }
480        } else {
481            size_based_chunk
482        }
483    }

485    /// Download a single chunk
486    async fn download_chunk(
487        client: Client,
488        url: &str,
489        chunk: ChunkInfo,
490        file: Arc<tokio::sync::Mutex<File>>,
491        _progress_tracker: Option<Arc<ProgressTracker>>,
492    ) -> Result<()> {
493        debug!(
494            "Downloading chunk {}: bytes {}-{}",
495            chunk.index, chunk.start, chunk.end
496        );
497
498        let range_header = format!("bytes={}-{}", chunk.start, chunk.end);
499        let response = client
500            .get(url)
501            .header("Range", range_header)
502            .send()
503            .await
504            .map_err(|e| TurboCdnError::network(format!("Failed to download chunk: {}", e)))?;
505
506        if !response.status().is_success() {
507            return Err(TurboCdnError::network(format!(
508                "Chunk download failed: {}",
509                response.status()
510            )));
511        }
512
513        let bytes = response
514            .bytes()
515            .await
516            .map_err(|e| TurboCdnError::network(format!("Failed to read chunk data: {}", e)))?;
517
518        // Write chunk to file
519        let mut file_guard = file.lock().await;
520        file_guard
521            .seek(SeekFrom::Start(chunk.start))
522            .await
523            .map_err(|e| TurboCdnError::io(format!("Failed to seek in file: {}", e)))?;
524
525        file_guard
526            .write_all(&bytes)
527            .await
528            .map_err(|e| TurboCdnError::io(format!("Failed to write chunk: {}", e)))?;
529
530        debug!("Completed chunk {}: {} bytes", chunk.index, bytes.len());
531        Ok(())
532    }

534    /// Download with single thread (fallback)
535    async fn download_single_thread<P: AsRef<Path>>(
536        &self,
537        url: &str,
538        output_path: P,
539        existing_size: u64,
540        _progress_tracker: Option<Arc<ProgressTracker>>,
541    ) -> Result<DownloadResult> {
542        info!("Starting single-threaded download");
543
544        let mut request = self.http_client.get(url);
545
546        // Add range header for resume if file exists
547        if existing_size > 0 {
548            request = request.header("Range", format!("bytes={}-", existing_size));
549        }
550
551        let response = request
552            .send()
553            .await
554            .map_err(|e| TurboCdnError::network(format!("Failed to start download: {}", e)))?;
555
556        if !response.status().is_success() {
557            return Err(TurboCdnError::network(format!(
558                "Download failed: {}",
559                response.status()
560            )));
561        }
562
563        // Open or create file
564        let mut file = if existing_size > 0 {
565            OpenOptions::new()
566                .append(true)
567                .open(output_path.as_ref())
568                .await
569                .map_err(|e| TurboCdnError::io(format!("Failed to open file: {}", e)))?
570        } else {
571            File::create(output_path.as_ref())
572                .await
573                .map_err(|e| TurboCdnError::io(format!("Failed to create file: {}", e)))?
574        };
575
576        // Stream download
577        let mut stream = response.bytes_stream();
578        let mut downloaded_bytes = existing_size;
579
580        use futures_util::StreamExt;
581        while let Some(chunk) = stream.next().await {
582            let chunk = chunk
583                .map_err(|e| TurboCdnError::network(format!("Failed to read chunk: {}", e)))?;
584
585            file.write_all(&chunk)
586                .await
587                .map_err(|e| TurboCdnError::io(format!("Failed to write chunk: {}", e)))?;
588
589            downloaded_bytes += chunk.len() as u64;
590        }
591
592        file.flush()
593            .await
594            .map_err(|e| TurboCdnError::io(format!("Failed to flush file: {}", e)))?;
595
596        Ok(DownloadResult {
597            path: output_path.as_ref().to_path_buf(),
598            size: downloaded_bytes,
599            duration: Duration::from_secs(0), // Will be set by caller
600            speed: 0.0,                       // Will be set by caller
601            url: url.to_string(),
602            resumed: existing_size > 0,
603        })
604    }
